# Base runtime requirements for the SEO platform backend.
#
# After the crawler-engine -> Django port, the crawler is synchronous and
# threaded; aiohttp / playwright / anthropic / psutil are no longer used.
# whitenoise + django-cors-headers stay because they are wired (or trivially
# wireable) into settings and are tiny.

Django>=4.2,<5
djangorestframework
psycopg2-binary
celery
redis
python-dotenv

# Crawler runtime
requests>=2.31
beautifulsoup4>=4.12
openpyxl>=3.1

# Windows-friendly TLS — makes urllib3/requests use the OS trust store
# instead of certifi's bundled CAs, which fixes CERTIFICATE_VERIFY_FAILED
# behind corporate MITM proxies. No-op on macOS / Linux.
truststore>=0.9

# Optional middleware (already referenced in design; kept for parity)
whitenoise>=6.6.0
django-cors-headers>=4.3.0

# ── SEO AI Agent System ─────────────────────────────────
# Groq is OpenAI-API-compatible, so we use the openai SDK with a
# different base_url instead of pulling a second LLM client. Pydantic
# powers the agent schemas; jsonschema validates LLM tool outputs.
openai>=1.40
pydantic>=2.5
jsonschema>=4.20

# Google Search Console API client. Used by:
#   - backend/scripts/gsc_pull.py (Search Analytics + sitemaps export)
#   - apps.crawler.storage.gsc_coverage_builder.upgrade_with_url_inspection
#     (per-URL verdicts via URL Inspection API)
# OAuth token lives in backend/data/gsc/ (gitignored).
google-api-python-client>=2.100
google-auth>=2.20
google-auth-oauthlib>=1.0

# Headless Chromium for real browser-console capture. Used by
# apps.crawler.engine.browser_console — captures live JS errors,
# uncaught exceptions, and failed network requests that the static
# `requests`-based crawler cannot see. Chromium binary is installed
# via `playwright install chromium` in the Dockerfile.
playwright>=1.40

# ── Scrapy engine (Phase 3 of the tool-clone roadmap) ───────────
# Activated via `python manage.py crawl_scrapy`. Runs alongside the
# legacy BFS engine behind the CRAWLER_ENGINE feature flag; both
# write to the same crawl_results.csv (legacy WAL) and to
# CrawlerPageResult in Postgres (Phase 3c dual-write). Scrapy brings
# AutoThrottle, RFPDupeFilter, RobotsTxtMiddleware, RetryMiddleware,
# DepthMiddleware — 15 years of crawler hardening for the cost of a
# `pip install`.
scrapy>=2.11
# JS rendering middleware. Triggered selectively in Phase 3e on pages
# whose static fetch returns < 500 chars of body text (SPA shells like
# iciciprulife homepage, which returned 19 words on static fetch).
scrapy-playwright>=0.0.40

# ── Phase 4 audit services ───────────────────────────────────────
# Internal PageRank ("Link Score") — same algorithm Ahrefs publishes,
# same algorithm Screaming Frog computes. networkx ships pure-Python
# pagerank(); 10k-node iteration converges in <2s on the existing
# crawl_discovered.csv link graph.
networkx>=3.0
# Near-duplicate detection via MinHash + LSH (Screaming Frog uses the
# same algorithm internally). datasketch's MinHashLSH is the de-facto
# Python implementation — 90% Jaccard threshold matches SF's default.
datasketch>=1.6

# ── Phase C SF parity dependencies ───────────────────────────────
# pypdf parses PDF metadata + text layer for the PDF-audit detectors
# (pdf_missing_title, pdf_scanned_no_text, pdf_encrypted). Pure-Python.
pypdf>=4.0
# pyspellchecker provides the small offline dictionary the readability
# detector keys off. Pure-Python, no external services, ~1MB dict.
pyspellchecker>=0.8

# Phase E — vendored axe-core for offline accessibility checks.
axe-core-python>=0.1.0

# ── TF-IDF keyword extraction ─────────────────────────────────────
# scikit-learn powers the in-house competitor content-keyword endpoint
# (TfidfVectorizer over the crawl corpus). Previously satisfied
# transitively via sentence-transformers / umap-learn — both of which
# were removed along with the embedding / 3D-content-map stack — so it
# is now pinned explicitly. Pure-Python feature; the caller degrades
# gracefully if the wheel is somehow absent.
scikit-learn>=1.3

# ── Subdomain → parent-brand consolidation ────────────────────────
# tldextract uses the Public Suffix List to derive the registrable
# domain (`auth.hdfclife.com` -> `hdfclife.com`). We pin the bundled
# snapshot via suffix_list_urls=() in apps.crawler.util.host so it
# never reaches out to the network at import time — the Bajaj corp
# proxy will block the publicsuffix.org fetch otherwise.
tldextract>=5.1
